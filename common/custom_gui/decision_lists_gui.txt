custom_button = {
	name = show_decision_list
	potential = {
		ai = no
	}
	effect = {
        if = {
			limit = { NOT = { has_country_flag = tm_show_decision_list } }
			set_country_flag = tm_show_decision_list
		}
		else = { clr_country_flag = tm_show_decision_list }
	}
	tooltip = "Open Toggle Viewing Decisions Menu"
}
