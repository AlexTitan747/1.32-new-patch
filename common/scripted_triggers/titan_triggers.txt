	custom_trigger_tooltip = {
		tooltip = trading_subject_in_$trade_company_region$_tt

		any_subject_country = {
			capital_scope = {
				trade_company_region = $trade_company_region$
			}
			is_subject_of = ROOT
			is_subject_of_type = trading_subject
			num_of_owned_provinces_with = {
				value = 15
				trade_company_region = $trade_company_region$
			}
		}
	}
}
