# country
on_startup = {
	if = {
		limit = {
			current_age = age_of_discovery
		}
		LXA = {
			if = {
				limit = {
					NOT = { has_country_flag = lxa_loyalty_flag }
				}
				add_estate_loyalty = {
					estate = estate_church
					loyalty = -10
				}
				set_country_flag = lxa_loyalty_flag
			}
		}
	}
	emperor = {
		save_global_event_target_as = Emperor
	}
	if = {
		limit = {
			is_emperor_of_china = yes
		}
		save_global_event_target_as = EmperorOfChina
	}
	events = {
		dif_ai.4
		muslim_school_events.20 #Pick School
#		flavor_fra.15000 #Make the French Revolution happen if starting in 1789
	}
	if = {
		limit = {
			has_dlc = "Cradle of Civilization"
			has_country_modifier = tur_janissary
		}
		remove_country_modifier = tur_janissary
	}
	initialize_schools_effect = yes #This is only used here but is used for readability of on_action file.

	if = {
		limit = {
			NOT = { has_country_flag = new_flavour_bav_13_has_been_triggered }
			OR = {
				tag = UBV
				tag = LBV
				tag = ING
			}
		}
		set_country_flag = new_flavour_bav_13_has_been_triggered
		country_event = { id = new_flavour_bav.13 days = 90 } #better not to allow almost instant DOWs
	}
	if = {
		limit = {
			tag = FRA
			started_in = 1789.7.14
			NOT = { is_year = 1790 }
			NOT = { has_disaster = french_revolution }
		}
		add_disaster_progress = {
			value = 100
			disaster = french_revolution
		}
	}
	if = {
		limit = {
			exists = SUK
		}
		SUK = {
			set_ruler_flag = suk_monarch_flag
		}
		AYU = {
			set_country_flag = ayu_monarch_flag
		}
	}
	every_country = {
		if = {
			limit = {
				check_variable = {
					which = admClicks
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = admClicks
				value = 0
			}
		}
		if = {
			limit = {
				check_variable = {
					which = dipClicks
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = dipClicks
				value = 0
			}
		}	
		if = {
			limit = {
				check_variable = {
					which = milClicks
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = milClicks
				value = 0
			}
		}
		if = {
			limit = {
				check_variable = {
					which = devsumClicks
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = devsumClicks
				value = 0
			}
		}
		if = {
			limit = {
				check_variable = {
					which = loanRepaid
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = loanRepaid
				value = 0
			}
		}
		if = {
			limit = {
				check_variable = {
					which = warsWon
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = warsWon
				value = 0
			}
		}
		if = {
			limit = {
				check_variable = {
					which = warsLost
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = warsLost
				value = 0
			}
		}
		if = {
			limit = {
				check_variable = {
					which = battlesWon
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = battlesWon
				value = 0
			}
		}
		if = {
			limit = {
				check_variable = {
					which = battlesLost
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = battlesLost
				value = 0
			}
		}
		if = {
			limit = {
				check_variable = {
					which = bankruptcyNum
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = bankruptcyNum
				value = 0
			}
		}
	}
}
