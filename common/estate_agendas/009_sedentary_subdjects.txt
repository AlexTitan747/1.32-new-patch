			factor = 0.65

						province_has_center_of_trade_of_level = 1

			}
		}
		modifier = {
			factor = 1.2
			any_owned_province = {
				is_state_core = root
				OR = {
					province_has_center_of_trade_of_level = 2
					province_trade_power = 20
				}
				NOT = { has_building = stock_exchange }
				OR = {
					AND = {

			factor = 2

			factor = 2
			NOT = {

			which = sedentary_subdjects_build_trade_building_in_y_var

		random_owned_province = {
			limit = {
				is_state_core = root
				OR = {
					province_has_center_of_trade_of_level = 2
					province_trade_power = 20
				}
				NOT = { has_building = stock_exchange }

					AND = {
						NOT = { has_building = marketplace }
						NOT = { has_building = trade_depot }
						num_free_building_slots = 1
					}
					AND = {
						ROOT = { dip_tech = 17 }
						NOT = { has_building = trade_depot }
						OR = {
							num_free_building_slots = 1
							has_building = marketplace
						}
					}
					AND = {
						ROOT = { dip_tech = 22 }
						OR = {
							num_free_building_slots = 1
							has_building = trade_depot
						}
					}
				}
			}
			save_event_target_as = agenda_province
		}
		event_target:agenda_province = {
			if = {
				limit = {
					has_building = trade_depot
				}
				ROOT = {
					set_variable = {
						which = sedentary_subdjects_build_trade_building_in_y_var
						value = 3
					}
				}
			}
			else_if = {
				limit = {
					has_building = marketplace

						value = 2
					}
				}

				ROOT = {
					set_variable = {
						which = sedentary_subdjects_build_trade_building_in_y_var
						value = 1
					}
				}
			}
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { owned_by = root }
		}
	}

		if = {
			limit = {
				check_variable = {
					which = sedentary_subdjects_build_trade_building_in_y_var

		}

			limit = {
				check_variable = {
					which = sedentary_subdjects_build_trade_building_in_y_var
					value = 2
				}
			}
			event_target:agenda_province = {
				OR = {
					has_building = trade_depot
					has_building = stock_exchange
				}
			}
		}
		else = {
			event_target:agenda_province = {
				OR = {
					has_building = marketplace
					has_building = trade_depot
					has_building = stock_exchange
				}
			}
		}
    }
    task_completed_effect = {
		add_estate_loyalty = {
			estate = sedentary_subdjects
			loyalty = 10
		}
		if = {
			limit = {
				NOT = { crown_land_share = 15 }
				estate_territory = {
					estate = sedentary_subdjects
					territory = 10
				}
			}
			take_estate_land_share_small = { estate = sedentary_subdjects }
		}
		else = {
			event_target:agenda_province = {
				add_base_production = 1
			}
		}
    }
    failing_effect = {
		event_target:agenda_province = {
			if = {
				limit = { owned_by = root }
				add_named_unrest = {
					name = local_sedentary_subdjects_displeased
					value = 5
				}
			}
		}
		if = {
			limit = { ai = no }
			add_estate_loyalty_modifier = {
				estate = sedentary_subdjects
				desc = EST_VAL_AGENDA_DENIED
				loyalty = -5
				duration = 7300
			}
		}
    }
}
