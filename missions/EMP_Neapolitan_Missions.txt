				limit = {
					religion = catholic

			}
			else = {

			}
		}
		effect = {
			if = {
				limit = {
					NOT = { owns_core_province = 118 }
				}
				add_papal_influence = 50
			}
			else = {
				add_prestige = 20
			}
		}
	}
	emp_nap_defender_of_the_faith = {
