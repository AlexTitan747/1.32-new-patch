					trade_goods = cloth
				}
			}
		}

	    effect = {
			every_owned_province = {
				limit = {
					region = crimea_region
					base_production = 10
					OR = {
						trade_goods = gems
						trade_goods = iron
						trade_goods = cloth
					}
				}
				if = {
					NOT = {
						has_production_building_trigger = yes
					}
					add_building = workshop
				}
				if = {
					NOT = {
						has_building = workshop
					}
					add_building = warehouse
				}
				add_base_production = 2
			}
	    }
	}
