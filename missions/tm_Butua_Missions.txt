	slot = 1
	generic = no
	ai = yes
	potential = {
		OR = {
			tag = RZW
			tag = ZBW
		}
	}
	has_country_shield = yes

	butua_renaissance = {
		icon = mission_fiqh_laws
		required_missions = {  }

		trigger = {
			OR = {
				has_institution = renaissance
				capital_scope = {
					development = 20
				}	
			}
		}

		effect = {
			if = {
				limit = {
					has_institution = renaissance
				}
				add_stability_or_adm_power = yes
			}
			else = {
				capital_scope = {
					add_institution_embracement = {
						which = renaissance
						value = 100
					}
				}
			}
        }
	}

	butua_tech_2 = {
		icon = reform_the_nomocanon
		required_missions = { butua_renaissance }

		trigger = {
			has_institution = renaissance
			OR = {
				has_completed_idea_group_of_category = { adm_ideas = yes }
				has_completed_idea_group_of_category = { dip_ideas = yes }
				has_completed_idea_group_of_category = { mil_ideas = yes }
			}
		}

		effect = {
			change_innovativeness = 3
			add_country_modifier = {
				name = "butua_technology_better"
				duration = 3650
			}
        }
	}
}
