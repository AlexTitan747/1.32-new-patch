oman_slot_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		tag = OMA
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	tm_oma_humilate_fars = {
		icon = the_hundred_years_war
		required_missions = { oma_buildforces }
		position = 2

		provinces_to_highlight = {
			OR = {
				area = farsistan_area
				area = persian_gulf_area
			}
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
		}

		trigger = {
				if = {
					limit = {
						OR = {
							exists = FRS
							NOT = {
								FRS = {
								is_subject_of = ROOT
								}
							}
							}
						}
					hidden_trigger = {
						OR = {
							has_country_flag = humiliated_rival
							has_country_flag = tm_show_strength
						}
					}
					is_rival = FRS
					else = {
						hidden_trigger = {
							OR = {
								has_country_flag = humiliated_rival
								has_country_flag = tm_show_strength
							}
						}
					}
				}
		}

		effect = {
			add_prestige = 15
			add_power_projection = 15
		}
	}
}
