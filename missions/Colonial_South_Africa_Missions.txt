colonial_SAF_slot_1 = { #Trade
    slot = 1
	generic = no
	ai = yes
	potential = {
		is_colonial_nation = yes
        capital_scope = {
            colonial_region = colonial_saf
        }
	}
	has_country_shield = yes

	saf_cape = {
		icon = needIcon
		required_missions = { }
		provinces_to_highlight = {
			province_id = 1177
		}

		trigger = {
			1177 = {
				owned_by = ROOT
				base_production = 10
			}
		}

		effect = {
			if = {
				limit = {
					1177 = {
						NOT = { province_has_center_of_trade_of_level = 3 }
					}
				}
				1177 = {
					add_center_of_trade_level = 1
				}
			}
			else = {
				1177 = {
					add_base_production = 3
				}
			}
		}
	}

	saf_cape_2 = {
		icon = needIcon
		required_missions = { saf_cape }
		provinces_to_highlight = {
			province_id = 1177
		}

		trigger = {
			1177 = {
				owned_by = ROOT
				base_production = 15
			}
		}

		effect = {
			if = {
				limit = {
					1177 = {
						NOT = { trade_goods = gold }
						NOT = { trade_goods = gems }
					}
				}
				1177 = {
					change_trade_goods = gems
				}
				
			}
			else = {
				1177 = {
					add_base_production = 3
				}
			}
		}
	}

	saf_cape_3 = {
		icon = needIcon
		required_missions = { saf_cape_2 }
		provinces_to_highlight = {
			province_id = 1177
		}

		trigger = {
			1177 = {
				owned_by = ROOT
			}
			1460 = {
				trade_share = {
					country = ROOT
					share = 50
				}
				is_strongest_trade_power = ROOT
			}
		}

		effect = {
			1177 = {
				add_province_modifier = {
					name = "cape_trade"
					duration = -1
				}
			}
		}
	}

	saf_cape_4 = {
		icon = needIcon
		required_missions = { saf_cape_3 }
		provinces_to_highlight = {
			province_id = 1177
		}

		trigger = {
			1177 = {
				owned_by = ROOT
			}
			1460 = {
				is_strongest_trade_power = ROOT
				trade_node_value = 25
			}
		}

		effect = {
			1460 = {
				add_trade_modifier = {
					who = ROOT
					duration = 5475
					power = 20
					key = STRONG_MERCHANTS
				}

		}

		trigger = {
			OR = {
				colonial_saf = {

					type = navigator

					religion = ROOT
				}

			else = {
				define_advisor = {
					type = treasurer
