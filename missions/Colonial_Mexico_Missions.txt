colonial_MEX_slot_1 = {
    slot = 1
	generic = no
	ai = yes
	potential = {
		is_colonial_nation = yes
        capital_scope = {
            colonial_region = colonial_mexico
        }
	}
	has_country_shield = yes

	mex_gold_1 = {
		icon = dominant_economy
		required_missions = { }
		provinces_to_highlight = {
			trade_goods = gold
			owned_by = ROOT
		}

		trigger = {
			OR = {
				any_owned_province = {
					trade_goods = gold
				}
				AND = {
					NOT = { gold = 1 }
					any_owned_province = {
						NOT = { province_id = 1177 }
						base_production = 6
					}
				}
			}
		}

		effect = {
			if = {
				limit = {
					gold = 1
				}
				random_owned_province = {
					limit = {
						trade_goods = gold
					}
					add_base_production = 3
				}
			}
			else_if = {
				limit = {
					NOT = { gold = 1 }
					any_owned_province = {
						NOT = { province_id = 1177 }
						base_production = 6
					}
				}
				random_owned_province = {
					limit = {
						is_city = yes
						NOT = { province_id = 1177 }
						base_production = 6
					}
					change_trade_goods = gold
					add_base_production = 1
				}
			}
			else = {
				custom_tooltip = saf_mission_gold_miss_tt
			}
		}
	}

	mex_gold_2 = {
		icon = chater_the_voc
		required_missions = { mex_gold_1 }

			}
		}
	}

	mex_gold_3 = {
		icon = mission_establish_mint_towns
		required_missions = { mex_gold_2 }
		provinces_to_highlight = {	}

		trigger = {
			OR = {
				any_owned_province = {
					trade_goods = gold
					base_production = 12
				}
				gold_income = 10
				gold_income_percentage = 0.12
			}
			
		}

		effect = {
			add_years_of_owned_provinces_production_income = {
				years = 4
				trigger = {
					trade_goods = gold
				}
				custom_tooltip = SAF_GOLD_GAIN
			}
		}
	}

	mex_gold_4 = {
		icon = reform_the_state
		required_missions = { mex_gold_3 }
		provinces_to_highlight = {	}

		trigger = {
			gold = 1
			NOT = { inflation = 5 }
		}

		effect = {
			custom_tooltip = saf_mission_gold_to_overlord_tt
			hidden_effect = {
				export_to_variable = {
					variable_name = saf_yearly_income
					value = years_of_income
				}
				while = {
					limit = {
						check_variable = {
							which = saf_yearly_income
							value = 1
						}
					}
					overlord = {
						add_treasury = 1
					}
					subtract_variable = {
						which = saf_yearly_income
						value = 1
					}
				}
			}
			add_country_modifier = {
				name = saf_exported_gold
				duration = 3650
			}

				estate_privilege = estate_burghers_control_over_the_mint
