#The Ottomans have installed Mengli Giray
country_event = {
	id = flavor_cri.6
	title = "flavor_cri.6.t"
	desc = "flavor_cri.6.d"
	picture = RELIGION_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		TUR = {
			has_country_flag = ottoman_promise_to_crimea
			NOT = { had_country_flag = { flag = ottoman_promise_to_crimea days = 23725 } } #Mengli was supposedly 30 when he fled to the Ottomans and 95 when he died.
		}
		tag = CRI
		is_subject_of = TUR
	}
	
	
	option = {
		name = "flavor_cri.6.a"
		ai_chance = {
			factor = 100
		}
		define_ruler  = {
			name = "Mengli"
			dynasty = "Giray"
			age = 33
			claim = 75
		}
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality_ancestor = { key = conqueror } }
		}
		add_opinion = {
			who = TUR
			modifier = installed_ruler_crimean_succession
		}
		TUR = {
			create_march = CRI
			hidden_effect = {
				country_event = { id = flavor_cri.7 }
			}
		}
		if = {
			limit = {
				285 = {
					owned_by = CRI
					is_capital = no
				}
			}
			285 = {
				cede_province = TUR
				add_core = TUR
			}
		}
		if = {
			limit = {
				286 = {
					owned_by = CRI
					is_capital = no
				}
			}
			286 = {
				cede_province = TUR
				add_core = TUR
			}
		}
		if = {
			limit = {
				2410 = {
					owned_by = CRI
					is_capital = no
				}
			}
			2410 = {
				cede_province = TUR
				add_core = TUR
			}
		}
		if = {
			limit = {
				2447 = {
					owned_by = CRI
					is_capital = no
				}
			}
			2447 = {
				cede_province = TUR
				add_core = TUR
			}
		}
	}
}

# Restoration of the Ilkhanate (alt history)
# by Daniel Tolman
country_event = {
	id = flavor_cri.8
	title = "flavor_cri.8.t"
	desc = "flavor_cri.8.d"
	picture = MONGOL_BANNERS_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	major = yes
	
	option = {
		name = flavor_cri.8.a
		persia_region = {
			limit = {
				owned_by = ROOT
			}
			add_core = ILK
			remove_core = ROOT
		}
		release = ILK
		create_march = ILK
		ILK = {
			change_government = monarchy
			#set_legacy_government = despotic_monarchy
			change_primary_culture = ROOT
			change_religion = ROOT
			define_ruler = {
				dynasty = ROOT
				culture = ROOT
				religion = ROOT
			}
		}
	}
	option = {
		name = flavor_cri.8.b
		4338 = {
			add_base_tax = 2
			add_base_manpower = 2
			add_base_production = 2
		}
	}
}
