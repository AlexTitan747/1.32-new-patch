		tag = BUR

				has_ruler = "Charles"

			factor = 0.5

		}

		name = "incidents_bur_inheritance.0.a"

			factor = 10
		}
		if = {
			limit = {
				is_year = 1473
			}

				name = "Marie"

		}
		else = {
			define_heir = {
				name = "Marie"
				dynasty = ROOT

				mil = 3
				claim = 95

	}
	option = {
		name = "incidents_bur_inheritance.0.b"
		ai_chance = { 
			factor = 10
		}
		add_legitimacy = 20
	}
}
