country_event = {
	id = titanslu.1
	title = "titanslu.title1"
	desc = "titanslu.desc1"
	picture = GOOD_WITH_MONARCH_eventPicture

	fire_only_once = yes
	
	trigger = {
		tag = SLU
		government = monarchy
		has_country_flag = formed_seleucid_flag
	}

	mean_time_to_happen = {
		months = 1
	}
	
	option = {
		name = "titanslu.ruler1"
			add_meritocracy_effect = yes
			define_ruler = {
				name = "Seleucus I"
				dynasty = "Nicator"
				culture = greek
				claim = 95
				adm = 3
				dip = 5
				mil = 3
				age = 22
				male = yes
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}	
			hidden_effect = { add_ruler_personality_ancestor = { key = strict } }	
		}
	}
}
