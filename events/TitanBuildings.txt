#cultural crisis
country_event = {
	id = tbuildings.5
	title = "tbuildings.5"
	desc = "tbuildings.desc5"
	picture = BUDDHA_eventPicture

	trigger = {
		has_country_flag = has_cultural_ministry

		dominant_culture = new_variable:domculture
		culture = new_variable:culture
		ruler_culture = new_variable:rulerculture
		heir_culture = new_variable:heirculture

		NOT = {
			OR ={
				is_variable_equal = {
					which = domculture
					which = culture
				}
				is_variable_equal = {
					which = rulerculture
					which = culture
				}
				is_variable_equal = {
					which = heirculture
					which = culture
				}
			}
		}
	}
	mean_time_to_happen = {
		months = 60
	}
	
	
	option = {		
	name = "tbuildings.e1"
		
	}

	option = {		
	name = "tbuildings.e2"
			
	}
}
