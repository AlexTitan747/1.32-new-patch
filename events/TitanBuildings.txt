province_event = {
	id =  tbuildings.50
	title = "tbuildings.50"
	desc = "tbuildings.desc50"
	picture = COLONIZATION_eventPicture
	
	trigger = {
		OR = {
			owner = { has_idea_group = exploration_ideas }
			owner = { has_idea_group = expansion_ideas }
			owner = { has_idea_group = colonialism_ideas }
			owner = { colonial_office = 1 }
		}
		has_empty_adjacent_province = yes
		any_empty_neighbor_province = {
			has_discovered = ROOT
			is_wasteland = no
			colonial_region_for_scope_province = {
				is_capital_of = ROOT
			}
		}
		owner = { NOT = { num_of_colonies = 2 } }
	}

	mean_time_to_happen = {
		months = 60
	}

	immediate = {
		hidden_effect = {
			random_empty_neighbor_province = {
				limit = {
					is_city = no
					has_colonist = no
					has_discovered = ROOT
					is_wasteland = no
					colonial_region_for_scope_province = {
						is_capital_of = ROOT
					}
				}
				save_event_target_as = empty_neighbor_target
			}
		}
	}
	
	option = {
		name = "tbuildings.i1"
		goto = empty_neighbor_target
		event_target:empty_neighbor_target = {
			create_colony = 200
			add_province_modifier = {
				name = "colonial_boom"
				duration = 7300
			}
		}
	}
	option = {
		name = "tbuildings.i2"
		trigger = {
			owner = {
			num_of_colonies = 1
			}
		}
		goto = root
		owner = {
			random_owned_province = {
				limit = {
					is_colony = yes
				}
				add_province_modifier = {
					name = "colonial_boom"
					duration = 1825
				}
				add_colonysize = 100
			}
		}
	}
}
