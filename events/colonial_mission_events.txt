country_event = { #SAF Gold
	id = colonial_mission_events.5
	title = colonial_mission_events.5.t
	desc = colonial_mission_events.5.d
	picture = REFORM_eventPicture
	
	is_triggered_only = yes

	option = {
		name = colonial_mission_events.1.a
        random_list = {
            5 = {
                random_owned_province = {
                    change_trade_goods = gold
                    add_base_production = 1
                }
                modifier = {
                    factor = 2
                    any_owned_province = {
                        development = 30
                    }
                }
            }
            95 = {
                add_prestige = -5
            }
        }
        add_country_modifier = {
            name = "colonial_subject_overlord_gold_reward"
            duration = -1
        }
        event_target:colonial_subject_event_target_for_rewards = {
            add_country_modifier = {
                name = "colonial_subject_gold_shared_reward"
                duration = -1
            }
        }
	}
	option = {
		name = colonial_mission_events.1.b
        event_target:colonial_subject_event_target_for_rewards = {
            add_country_modifier = {
                name = "colony_gold"
                duration = -1
            }
            random_owned_province = {
                limit = {
                    trade_goods = gold
                }
                add_base_production = 2
            }
        }
	}
}
