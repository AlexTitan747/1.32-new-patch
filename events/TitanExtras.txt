namespace = extras

#applies exodus penalties
country_event = {
	id = extras.1
	title = "extras.1"
	desc = "extras.desc1"
	picture = BUDDHA_eventPicture

	trigger = {
	NOT = {
		OR = {
		technology_group = north_american
		technology_group = mesoamerican
		technology_group = south_american
		technology_group = andean
		technology_group = high_american
		}
		OR = {
		has_country_flag = exodus_penalties_applied
		tag = PBR
		}
	}
	capital_scope = {
					OR = {
						continent = new_world
						continent = north_america
						continent = south_america
					}
				}
	is_year = 1444
	is_colonial_nation = no
}
	mean_time_to_happen = {
		months = 1
	}
	
	
option = {		
	name = "extras.true1"
	add_country_modifier = { 
			name = exodus_nation
			duration = -1
			#hidden = no
	}
	set_country_flag = exodus_penalties_applied
}
}

country_event = {
	id = extras.7
	title = "titanextras.title7"
	desc = "titanextras.desc7"
	picture = RELIGION_eventPicture

	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			AND = {
				tag = HES
				was_tag = HES
			}
		}
		KSM = {
			has_country_flag = hussite_missionaries_sent
		}
	}

	mean_time_to_happen = {
		months = 1
	}

	option = {
		name = "titanextras.religion1"
		change_religion = hussite
		add_country_modifier = {
			name = converted_hussite
			duration = 7300
		}
		1876 = {
			change_religion = hussite
			add_reform_center = hussite
		}
	}

	option = {
		name = "titanextras.religion2"
		add_prestige = 20
		add_country_modifier = {
			name = stayed_true_to_our_faith
			duration = 3650
		}
	}
}
