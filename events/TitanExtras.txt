namespace = extras

#applies exodus penalties
country_event = {
	id = 	extras.1
	title = "extras.1"
	desc = "extras.desc1"
	picture = BUDDHA_eventPicture

	trigger = {
	
	NOT = {
		OR = {
		technology_group = north_american
		technology_group = mesoamerican
		technology_group = south_american
		technology_group = andean
		technology_group = high_american
		}
	}
	capital_scope = {
					OR = {
						continent = new_world
						continent = north_america
						continent = south_america
					}
				}
    has_country_flag = exodus_penalties_applied
	is_year = 1444
}
	mean_time_to_happen = {
		months = 1
	}
	
	
option = {		
	name = "extras.true1"
	add_country_modifier = { 
			name = exodus_nation
			duration = -1
			hidden = no
			}
	set_country_flag = exodus_penalties_applied
}
}

country_event = {
	id = 	extras.2
	title = "extras.2"
	desc = "extras.desc2"
	picture = COURT_eventPicture

	trigger = {
	
	ai = no
	NOT = {
		any_owned_province = {
				has_province_flag = titan_free_banner
			}
		}
		any_owned_province = {
			culture = ROOT
			is_core = ROOT
			is_state_core = root
		}
	
}
	mean_time_to_happen = {
		days = 5
	}
	
	
option = {		
	name = "extras.true2"
	random_owned_province = {
			limit = {
				culture = ROOT
				is_core = ROOT
				is_state_core = root
			}
			set_province_flag = titan_free_banner
			add_province_triggered_modifier = titan_free_banner
		}	
			
			set_country_flag = freebanners_got
}
}

country_event = {
	id = extras.3
	title = "extras.3"
	desc = "extras.desc3"
	picture = COURT_eventPicture
	
	trigger = {
		num_of_owned_provinces_with = {
			value = 2
			has_province_flag = titan_free_banner
		}
	}
	mean_time_to_happen = {
		days = 5
	}
	option = {
		name = "extras.toomany"
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_flag = titan_free_banner
				}
				clr_province_flag = titan_free_banner
				remove_province_triggered_modifier = titan_free_banner
				
			}
		}
	}
}
