country_event = {
	id = chinagov.1
	title = "chinagov.title1"
	desc = "chinagov.desc1"
	picture = BUDDHA_eventPicture

	trigger = {
	OR = {
		tag = JNG 
		tag = ZHJ
		tag = FUA
		tag = GDO
		tag = GXI
		tag = HNA
		tag = JIX
		tag = AUI
		tag = SAN
		tag = LNI
		tag = HEN
		tag = HEI
		tag = SCH
		tag = SXI
		tag = YNA
		tag = ALI
		tag = GSU
		tag = ORO
		tag = SHU
		tag = WEI
		tag = WUU
		}
		is_year = 1444
		NOT = {
		has_country_flag = chinagov_duchy
		}
	}

	mean_time_to_happen = {
		months = 1
	}
	
option = {		
		name = "chinagov.king1"
		if = {
			limit = {
				NOT = {
				has_reform = warring_state_reform
				}
			}
			add_government_reform = warring_state_reform
		}
		hidden_effect = {
		set_country_flag = chinagov_duchy
		add_country_modifier = { 
				name = china_fractured
				duration = -1
				hidden = no
			}
		}
}
option = {		
		name = "chinagov.repu1"
		remove_government_reform = warring_state_reform
		change_government_to_republic = yes
		add_government_reform = chinese_merchant_state_reform 
		hidden_effect = {
		set_country_flag = chinagov_duchy

			name = china_fractured
			duration = -1
			hidden = no
			}
		}
}
